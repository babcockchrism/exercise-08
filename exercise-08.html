For generating a null sampling distribution for the slope coefficient, I would need to permute the ECV and group size each.

nperm <- 1000
permuted_beta1 <- vector(length = nrow(d_ECVvsGS))
for(i in 1:nperm) {
  x <- sample(d_ECVvsGS$ECV)
  y <- sample(d_ECVvsGS$Group_size)
  permuted_beta1[[i]] <- cov(x, y)/var(y)

n_boot <- 10000
boot <- do(n_boot) * mean(~sample(s, length(s), replace = TRUE))

theory <- mean(boot) + c(-1, 1) * qnorm(0.975) * se
